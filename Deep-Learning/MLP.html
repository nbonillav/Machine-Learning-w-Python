[25 puntos] En este ejercicio se usará la tabla de datos students_dropout.csv. Este conjunto de datos contiene datos de una institución de educación superior sobre diversas variables relacionadas con los estudiantes de pregrado, para investigar el impacto de estos factores en la deserción de los estudiantes y el éxito académico. La tabla contiene 3630 filas y 13 columnas, las cuales se explican a continuación.

  • marital_status: Estado Civil. (Categórica)
  • daytime_evening_attendance: Si el estudiante asiste a clases durante el día o la noche. (Categórica)
  • educational_special_needs: Si el estudiante tiene alguna necesidad educativa especial. (Categórica)
  • debtor: Si el estudiante es deudor. (Categórica)
  • tuition_fees_up_to_date: Si las tasas de matrícula del estudiante están al día. (Categórica)
  • gender: Género del estudiante. (Categórica)
  • scholarship : Si el estudiante tiene beca. (Categórica)
  • age_at_enrollment: Edad del estudiante al momento de la inscripción. (Numérica)
  • international: Si el estudiante es extranjero. (Categórica)
  • unemployment_rate: Tasa de desempleo. (Numérica)
  • inflation : Tasa de inación. (Numérica)
  • gdp: PIB per cápita. (Numérica)
  • target: Indica si el estudiante se graduó o abandonó. (Categórica, Variable a predecir)

def asCategoryType(df, asCategory=True):
    df_copy = df.copy()
    target_var = df_copy['target']
    df_copy = df_copy.drop(columns=('target'))
    if asCategory:
        num_df = df_copy.select_dtypes(include=[np.number])
        cat_df = df_copy.select_dtypes(exclude=[np.number])

    if not cat_df.empty:
        for i in cat_df:
            cat_df[i] = cat_df[i].astype('category')
            df_copy = pd.concat([num_df, cat_df, target_var], axis=1)
    else:
        df_copy = pd.concat([num_df, cat_df], axis=1)
    
    return df_copy

def distribucion_variable_predecir(data:pd.DataFrame, variable_predict:str):
  conteo     = data[variable_predict].value_counts()
  valores    = conteo.to_list()
  categorias = conteo.index.to_list()
  titulo     = "Distribución de la variable %s" % variable_predict
  
  fig, ax = plt.subplots()
  ax.bar(categorias, valores)
  
  ax.set_ylabel(variable_predict)
  ax.set_title(titulo)
  
  for i, valor in enumerate(valores):
    porc = round(valor / sum(valores) * 100, 2)
    text = str(valor) + "\n" + str(porc) + "%"
    ax.text(i, valor * 0.45, text, color='white', ha='center', fontsize=10)
  
  return(fig)

def indices_general(MC, nombres = None):
  precision_global = np.sum(MC.diagonal()) / np.sum(MC)
  error_global     = 1 - precision_global
  precision_categoria  = pd.DataFrame(MC.diagonal()/np.sum(MC,axis = 1)).T
  if nombres!=None:
    precision_categoria.columns = nombres
    return {"Matriz de Confusión":MC, 
            "Precisión Global":   precision_global, 
            "Error Global":       error_global, 
            "Precisión por categoría":precision_categoria}

Modelo 1¶

Para agregar una capa debemos utilizar la función Dense. Luego podemos definir los siguientes parámetros:

  • units: $10, 5$ -> Define la cantidad de nodos.
  • activation: $relu, sigmoid$ -> Define la función de activación a utilizar.
    Estos pueden ser: deserialize, elu, exponential, gelu, get, hard_sigmoid, linear, mish, relu, selu, serialize, sigmoid, softmax, softplus, softsign, swish y tanh.

Al final siempre debemos configurar el método compile. Los parámetros pueden ser:

  • optimizer: $adam$ -> Define el solucionador.
    pueden ser: adadelta, adafactor, adagrad, adam, adamW, adamax, ftrl, lion, nadam, optimizer y sgd.

  • loss: binary_crossentropy -> Define la función de perdida a utilizar.

class redNeuronal():
    def __init__(self, entrada, pesos, umbral, z):
        self._x = np.array(entrada)
        self._w = np.array(pesos)
        self._t = np.array(umbral)
        self._z = z                   # Valors Reales
        self._y = None                # Prediccion
        self._mse = None              # Error cuadratico Medio
        self._a = None
        self._activacion = None
        self._fn = None
    
    # get
    @property
    def x(self):
        return np.array(self._x)
    @property
    def w(self):
        return np.array(self._w)
    @property
    def t(self):
        return np.array(self._t)
    @property
    def z(self):
        return self._z
    @property
    def a(self):
        return self._a
    @property
    def y(self):
        return self._y
    @property
    def activacion(self):
        return self._activacion
    @property
    def mse(self):
        return self._mse
    @property
    def fn(self):
        return self._fn
    
    # Methods
    # -------------------------------------------------
    # Cálculo del perceptrón: a = w·x - t
    # -------------------------------------------------
    def calc_perceptron(self):
        self._a = np.sum(self._x * self._w) - self._t  # aseguramos arrays
        self._activacion = self._a  # guardamos activación
        return self._a
    
    # -------------------------------------------------
    # funcion de activacion | SoftSign
    # -------------------------------------------------
    def softsign(self, a):
        self._fn = a / (1 + np.abs(a)) # funcion de activacion
        return self._fn

    # -------------------------------------------------
    # Función indicadora I(t)
    # -------------------------------------------------
    # se asigna 1 o 0 a la variable de prediccion y
    def I(self): # Indicador I(t)
        if self._fn >= 0:
            self._y = 1
        else:
            self._y = 0
        return self._y


    # -------------------------------------------------
    # Pipeline completo
    # -------------------------------------------------  
    def calculos(self):
        a = self.calc_perceptron()
        self.softsign(a) # f(a) -> aplicar la funcion de activacion, salida suave. f(a)
        self.I()
        return self._y
    

    # -------------------------------------------------
    # Comparación con valor real
    # -------------------------------------------------
    # comparamos la prediccion y con el valor z (valor correcto)
    def imp_comparacion_con_prediccion(self):
        if self._y is None:
            self.calculos()  # aseguramos que se haya ejecutado los calculos
        
        if self._y == self._z:
            print("\nBien clasificado")
            print('Porque t:', round(self._activacion, 2), '>= 0')
            print('Entonces, y:',self._y, '== z:',self._z)
        else:
            print("\nMal clasificado")
            print('Porque t:', round(self._activacion, 2), '< 0')
            print('Entonces, y:',self._y, '!= z:',self._z)


    # -------------------------------------------------
    # Error cuadrático para un patrón
    # -------------------------------------------------     
    def indice_mse(self):
        # Error cuadrático
        self._mse = (self._y - self._z)**2
        return self._mse
